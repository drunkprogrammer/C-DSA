#include<string>
class Solution {
public:
    string longestPalindrome(string s) {
        string rs=s;
        reverse(s.begin(),s.end());
        int size=s.length();
        int dynamictable[size][size];
        int longest=0;
        int start;
        int start_s;
        int start_r;
        for(int i=0;i<size;i++)
        {
            dynamictable[0][i]=(s[0]==rs[i])?1:0;
        }
         for(int i=1;i<size;i++)
        {
            dynamictable[i][0]=(rs[0]==s[i])?1:0;
            for(int j=1;j<size;j++)
            {
                    dynamictable[i][j]=(s[i]==rs[j])?dynamictable[i-1][j-1]+1:0;
            }
        }
        for(int i=0;i<size;i++)
        {
            for(int j=0;j<size;j++)
            {
                if(dynamictable[i][j]>longest)
                {
                    start_s=i-dynamictable[i][j]+1;
                    start_r=j-dynamictable[i][j]+1; 
                    if((start_s+start_r+1)!=size)
                    {
                      continue;
                    }
                    else
                    {
                         
                        longest=dynamictable[i][j];
                        start=j-longest+1;
                    }
                }
            }
        }
      return   rs.substr(start,longest);
    }
};
