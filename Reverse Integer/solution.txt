        int decade;
        int hundred;

        decade=(x/10)%10;
        hundred=x/100;

        {
            if(decade==0)
            {
                if(hundred==0)
                {
                    result=0;
                }
                else
                {
                    result=hundred;
                }
            }
            else
            {
                 if(hundred==0)
                {
                    result=decade;
                }
                else
                {
                    result=decade*10+hundred;
                }
            }
        }
        else
        {
            if(decade==0)
            {
                if(hundred==0)
                {
                    result=unit;
                }
                else
                {
                    result=unit*100+hundred;
                }
            }
            else
            {
                 if(hundred==0)
                {
                    result=unit*10+decade;
                }
                else
                {
                    result=unit*100+decade*10+hundred;
                }
            } 
        }
            return result;
